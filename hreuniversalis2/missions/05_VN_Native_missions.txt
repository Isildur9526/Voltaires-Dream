VN_tribal_missions = {
	slot = 1
	generic = yes
	ai = yes
	potential = {
		government = native
		NOT = { map_setup = map_setup_random }
	}
	
	vn_tribal_federation = {
		icon = mission_noble_council
		required_missions = { }
		trigger = {
			if = {
				limit = { has_dlc = "Conquest of Paradise" primitives = yes }
				is_federation_leader = yes
			}
			else = {
				num_of_allies = 4
			}
			diplomatic_reputation = 3
		}
		effect = {
			add_country_modifier = {
				name = "tribal_federation_mission"
				duration = 9125
			}
		}
	}
	vn_expand_tribal_territory = {
		icon = mission_tribal_war
		required_missions = { vn_tribal_federation }
		trigger = {
			capital_scope = {
				area_for_scope_province = {
					type = all
					owned_by = ROOT
				}
			}
			NOT = {
				any_neighbor_country = {
					army_size = ROOT
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "expand_tribal_territory_mission"
				duration = 9125
			}
		}
	}
	vn_tribal_capital = {
		icon = mission_tribe
		required_missions = { vn_expand_tribal_territory }
		provinces_to_highlight = {
			is_capital_of = ROOT
		}
		trigger = {
			if = {
				limit = { has_dlc = "Conquest of Paradise" primitives = yes }
				calc_true_if = {
					desc = vn_tribal_capital.tt
					capital_scope = {
						has_building = native_irrigation
						has_building = native_fortified_house
						has_building = native_ceremonial_fire_pit
						has_building = native_sweat_lodge
						has_building = native_longhouse
						has_building = native_earthwork
						has_building = native_great_trail
						has_building = native_palisade
						has_building = native_storehouse
						has_building = native_three_sisters_field
					}
					amount = 5
				}
			}
			else = {
				capital_scope = {
					base_tax = 4
					base_production = 4
					base_manpower = 4
					NOT = { unrest = 1 }
					NOT = { devastation = 1 }
				}
			}
		}
		effect = {
			if = {
				limit = { has_dlc = "Conquest of Paradise" primitives = yes }
				capital_scope = {
					add_base_tax = 2
				}
			}
			add_country_modifier = {
				name = "tribal_capital_mission"
				duration = 9125
			}
		}
	}
	vn_introduce_feudalism = {
		icon = mission_coronation
		required_missions = { vn_tribal_capital }
		trigger = {
			has_institution = feudalism
			all_state_province = {
				feudalism = 100
			}
			NOT = { government = native }
		}
		effect = {
			swap_non_generic_missions = yes
			add_country_modifier = {
				name = "introduce_feudalism_mission"
				duration = 9125
			}
		}
	}
}
