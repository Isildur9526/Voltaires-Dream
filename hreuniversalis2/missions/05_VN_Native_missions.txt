VN_tribal_missions = {
	slot = 1
	generic = yes
	ai = yes
	potential = {
		government = native
		NOT = { map_setup = map_setup_random }
	}
	
	vn_tribal_federation = {
		icon = mission_noble_council
		required_missions = {}
		trigger = {
			if = {
				limit = { has_dlc = "Conquest of Paradise" primitives = yes }
				is_federation_leader = yes
			}
			else = {
				num_of_allies = 4
			}
			diplomatic_reputation = 3
		}
		effect = {
			add_country_modifier = {
				name = "tribal_federation_mission"
				duration = 9125
			}
		}
	}
	vn_expand_tribal_territory = {
		icon = mission_tribal_war
		required_missions = { vn_tribal_federation }
		trigger = {
			capital_scope = {
				area_for_scope_province = {
					type = all
					owned_by = ROOT
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "expand_tribal_territory_mission"
				duration = 9125
			}
		}
	}
	vn_tribal_capital = {
		icon = mission_tribe
		required_missions = { vn_expand_tribal_territory }
		trigger = {
			if = {
				limit = { has_dlc = "Conquest of Paradise" primitives = yes }
				capital_scope = {
					has_building = native_irrigation
					has_building = native_fortified_house
					has_building = native_ceremonial_fire_pit
					has_building = native_sweat_lodge
					has_building = native_longhouse
				}
			}
			else = {
				capital_scope = {
					base_tax = 6
					base_production = 6
					base_manpower = 6
					num_of_times_improved = 10
					NOT = { devastation = 1 }
				}
			}
		}
		effect = {
			if = {
				limit = { has_dlc = "Conquest of Paradise" primitives = yes }
				capital_scope = {
					add_base_tax = 1
					add_base_production = 1
					add_base_manpower = 1
				}
			}
			add_country_modifier = {
				name = "tribal_capital_mission"
				duration = 9125
			}
		}
	}
	vn_introduce_feudalism = {
		icon = mission_coronation
		required_missions = { vn_tribal_capital }
		trigger = {
			has_institution = feudalism
			all_state_province = {
				feudalism = 100
			}
			NOT = { government = native }
		}
		effect = {
			add_country_modifier = {
				name = "introduce_feudalism_mission"
				duration = 9125
			}
		}
	}
}
