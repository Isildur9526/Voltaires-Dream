VN_crusader_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		OR = {
			religion = catholic
			religion = catholique
		}
		OR = {
			tag = TEU
			tag = LIV
			tag = KNI
			capital_scope = {
				superregion = balkans_superregion
			}
			capital_scope = {
				superregion = levant_superregion
			}
		}
		NOT = { culture_group = albanian_group }
		NOT = { culture_group = south_slavic }
		NOT = { map_setup = map_setup_random }
	}
	
	vn_increase_recruitment = {
		icon = mission_soldiers_in_line
        required_missions = {} 
        position = 5
		trigger = {
			army_size_percentage = 1
			manpower_percentage = 0.8
			armory = 5
		}
		effect = {
			add_country_modifier = {
				name = "increase_recruitment_mission"
				duration = 9125
			}
		}
	}
	vn_unity_of_the_crusaders = {
		icon = mission_religious
		required_missions = { vn_increase_recruitment } 
		trigger = {
			check_variable = { which = num_converted_religion value = 10 }
			religious_unity = 0.5
		}
		effect = {
			add_country_modifier = {
				name = "unity_of_the_crusaders_mission"
				duration = 9125
			}
		}
	}
	vn_templar_forts = {
		icon = mission_fort
		required_missions = { vn_unity_of_the_crusaders } 
		trigger = {
			if = {
				limit = {
					has_dlc = "The Cossacks"
					OR = {
						has_estate = estate_cossacks
						current_age = age_of_empire
					}
				}
				num_of_owned_provinces_with = {
					value = 5
					has_fort_trigger = yes
				}
				estate_influence = {
					estate = estate_cossacks
					influence = 60
				}
				estate_loyalty = {
					estate = estate_cossacks
					loyalty = 70
				}
			}
			else = {
				num_of_owned_provinces_with = {
					value = 10
					has_fort_trigger = yes
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "templar_forts_mission"
				duration = 9125
			}
		}
	}
	vn_crusader_army = {
		icon = mission_teutonic_knights
		required_missions = { vn_templar_forts } 
		trigger = {
			num_of_generals = 3
			if = {
                limit = {
					OR = {
						has_estate = estate_holy_orders
						current_age = age_of_empire
					}
				}
				custom_trigger_tooltip = {
					tooltip = vn_support_of_the_templars.tt
					calc_true_if = {
						all_owned_province = {
							has_estate = estate_holy_orders
						}
						amount = 10
					}
				}
				estate_loyalty = {
					estate = estate_holy_orders
					loyalty = 70
				}
            }
			num_of_cavalry = 10
		}
		effect = {
			add_country_modifier = {
				name = "crusader_army_mission"
				duration = 9125
			}
		}
	}
}
