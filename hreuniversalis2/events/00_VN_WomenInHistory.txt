# Anna Komnena (1083 - 1154)
country_event = {
	id = vn_women_in_history.1
	title = vn_women_in_history.1.t
	desc = vn_women_in_history.1.d
	picture = wihgfx_FEMALE_RULER_eventPicture
	
	fire_only_once = yes

	trigger = {
		is_year = 1083
		NOT = {
			is_year = 1154
		}
		tag = BYZ
		is_female = no
		has_regency = no
		has_heir = no
		is_lesser_in_union = no
		government = monarchy
		has_reform = eastern_monarchy
		has_country_flag = komnenian_restoration_flag
		NOT = {
			advisor_exists = 2090
			is_advisor_employed = 2090
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = vn_women_in_history.1.a
		define_heir = {
			name = "Anna"
			dynasty = "Komnenos"
			culture = greek
			adm = 5
			dip = 3
			mil = 3
			age = 23
			female = yes
		}

		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_heir_personality = scholar_personality }
		}
	}
	option = {
		name = vn_women_in_history.1.b
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_meritocracy_effect = yes
		define_advisor = {
			culture = greek
			female = yes
			type = artist
			skill = 2
			discount = yes
			name = "Anna Komnenos"
		}
	}
