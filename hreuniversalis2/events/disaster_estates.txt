# The Nobles take control
country_event = {
	id = estate_disasters.1
	title = estate_disasters.1.t
	desc = estate_disasters.1.d
	picture = NOBLE_ESTATE_TAKE_OVER_eventPicture
	
	major = yes
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = noble_estate_in_power
	}
	
	option = {
		name = estate_disasters.1.a	
		add_stability = -1
		if = {
			limit = {
				has_reform = elective_monarchy
			}
			add_country_modifier = {
				name = noble_power_elective_monarchy
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_reform = prussian_monarchy
			}
			add_country_modifier = {
				name = junker_dominance_prussian_monarchy
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_reform = ottoman_government
			}
			add_country_modifier = {
				name = umera_dominance_ottoman_government
				duration = -1
			}
		}
		else_if = {
			limit = {
				OR = {
					has_reform = tsardom
					has_reform = principality
				}
			}
			add_country_modifier = {
				name = boyar_dominance
				duration = -1
			}
		}
		else_if = {
			limit = { is_emperor_of_china = yes }
			reduce_meritocracy_large_effect = yes
			add_country_modifier = {
				name = gentry_dominance
				duration = -1
			}
		}
		else_if = {
			limit = {
				OR = {
					has_reform = iqta
					has_reform = feudal_theocracy
					has_reform = sufi_reform
					has_reform = tribal_federation
				}
			}
			add_country_modifier = {
				name = amir_dominance
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_reform = mamluk_government
			}
			add_legitimacy = -10
			define_ruler = { culture = circassian }
			add_country_modifier = {
				name = mamluk_amir_dominance
				duration = -1
			}
		}
		else = {
			add_country_modifier = {
				name = "noble_power_monarchy"
				duration = -1
			}
		}
		custom_tooltip = estate_disasters.1.a.tt
	}
}

# The Mamluks revolt!
country_event = {
	id = estate_disasters.21
	title = estate_disasters.21.t
	desc = estate_disasters.21.d
	picture = ANGRY_MOB_eventPicture
	
	major = yes
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = mamluks_estate_triggered
	}
	
	option = {
		name = estate_disasters.21.a
		ai_chance = { factor = 1 }
		create_independent_estate = {
			estate = estate_mamluks
			government = monarchy
			government_reform = mamluk_government
			custom_national_ideas = MAM_ideas
			play_as = no
		}
	}
	
	option = {
		name = estate_disasters.21.b
		trigger = { 
			ai = no 
			has_dlc = "The Cossacks"
		}
		create_independent_estate = {
			estate = estate_mamluks
			government = monarchy
			government_reform = mamluk_government
			custom_national_ideas = MAM_ideas
			play_as = yes
		}
	}

	option = {
		name = estate_disasters.21.c
		trigger = { NOT = { has_reform = mamluk_government } }
		ai_chance = {
			factor = 99
			modifier = {
				factor = 0
				NOT = { tag = MAM }
			}
		}
		change_government = monarchy
		add_government_reform = mamluk_government
		if = {
			limit = { primary_culture = hejazi_culture }
			define_ruler = {
				dynasty = "Labiyya"
				culture = old_egyptian
			}
		}
		if = {
			limit = { primary_culture = bedouin_arabic }
			define_ruler = {
				dynasty = "Najahiyya"
				culture = aramaic
			}
		}
		if = {
			limit = { primary_culture = gulf_arabic }
			define_ruler = {
				dynasty = "Jannabiyya"
				culture = persian
			}
		}
		if = {
			limit = { primary_culture = moroccan }
			define_ruler = {
				dynasty = "Barbariyya"
				culture = tuareg
			}
		}
		if = {
			limit = { primary_culture = algerian }
			define_ruler = {
				dynasty = "Qulaughliyya"
				culture = greek
			}
		}
		if = {
			limit = { primary_culture = tunisian }
			define_ruler = {
				dynasty = "Armaniyya"
				culture = armenian
			}
		}
		if = {
			limit = { primary_culture = al_misr_arabic }
			define_ruler = {
				dynasty = "Bahriyya"
				culture = turkish
			}
		}
		if = {
			limit = { primary_culture = al_iraqiya_arabic }
			define_ruler = {
				dynasty = "Jawrajiyya"
				culture = georgian
			}
		}
		if = {
			limit = { primary_culture = andalucian }
			define_ruler = {
				dynasty = "Saqalibiyya"
				culture = croatian
			}
		}
		if = {
			limit = { primary_culture = al_suryah_arabic }
			define_ruler = {
				dynasty = "Burjiyya"
				culture = circassian
			}
		}
		hidden_effect = {
			remove_heir = yes
			swap_free_idea_group = yes
			if = {
				limit = { tag = MAM }
				override_country_name = MAMLUKS_NAME
				set_country_flag = mamluks_name_flag
			}
		}
	}
}
