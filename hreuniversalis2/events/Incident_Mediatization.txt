#Initial Event
country_event = {
	id = incident_mediatization.1
	title = incident_mediatization.1.t
	desc = incident_mediatization.1.d
	picture = HRE_eventPicture
	
	fire_only_once = yes 
	
	trigger = {
		is_institution_enabled = enlightenment
		num_of_foreign_hre_provinces = 100
		NOT = { started_in = 1789.7.14 }
		#NOT = { active_imperial_incident = any } #add - uncomment in 1.30
		is_part_of_hre = no
		is_at_war = no
		NOT = {
			has_country_flag = med_demander_country
		}
		any_country = {
			is_part_of_hre = yes
			government = theocracy
		}
		any_country = {
			is_part_of_hre = yes
			has_reform = free_city
		}
		any_owned_province = {
			is_part_of_hre = yes
		}
		NOT = {
			any_known_country = {
				is_part_of_hre = no
				any_owned_province = {
					is_part_of_hre = yes
				}
				num_of_cities = ROOT
			}
		}
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	immediate = {
		hidden_effect = {
			save_event_target_as = med_demander
			set_country_flag = med_demander_country
		}
	}
	
	option = {	#Demand
		name = incident_mediatization.1.a
		ai_chance = { factor = 100 }
		custom_tooltip = incident_mediatization.1.a.tt
		hidden_effect = {
			every_country = {
				limit = {
					is_part_of_hre = yes
					NOT = { tag = PAP }
					OR = {
						government = theocracy
						has_reform = free_city
					}
				}
				country_event = {
					id = incident_mediatization.2	#All theocracies/free cities informed of their fate
					days = 20
					random = 10
				}
			}
		}
		#add - replace this block with set_imperial_incident = incident_mediatization
		#### start of block
		emperor = {
			country_event = {
				id = incident_mediatization.3	#incident event
				days = 60
				random = 10
			}
		}
		#### end of block
	}
	
	option = {	#Back down
		name = incident_mediatization.1.b
		ai_chance = { factor = 0 }
		add_prestige = -20
	}
}

								}
							}

								limit = {
									owned_by = ROOT
								}

									limit = { owned_by = ROOT }
									remove_core = ROOT
									cede_province = event_target:mediatize_theo_country
									add_core = event_target:mediatize_theo_country
