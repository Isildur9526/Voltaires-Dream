#Events for the Almoravid invasion of Al-Andalus in the late eleventh century 

namespace = almoravids_events

#Turns DAA into MOR
country_event = {
	id = almoravids_events.1
    title = "almoravids_events.1.t"
    desc = "almoravids_events.1.d"
    picture = RELIGIOUS_WARS_eventPicture

	fire_only_once = yes
	major = yes

	mean_time_to_happen = {
		months = 60
	}

	trigger = {
		normal_or_historical_nations = yes
		NOT = { started_in = 1061.1.1 }
		is_year = 1057
        NOT = { is_year = 1080 }
		tag = DAA
        dynasty = "al-Murâbit"
        religion = sunni
        has_regency = no
        is_at_war = no
        NOT = { exists = TFL }
	}

	option = {
		name = almoravids_events.1.a
		change_tag = MOR
		if = {
			limit = { MRK = { ai = yes was_player = no } }
			inherit = MRK
		}
		if = {
			limit = { TDL = { ai = yes was_player = no } }
			inherit = TDL
		}
		if = {
			limit = { SOS = { ai = yes was_player = no } }
			inherit = SOS
        }
		hidden_effect = {
			set_government_rank = 5
			change_government = monarchy
			add_government_reform = iqta
			set_capital = 4762
            add_accepted_culture = moroccan
            override_country_name = ALMORAVIDS_NAME
            set_country_flag = almoravids_name_flag
            add_treasury = 200
            add_stability = 3
            add_prestige = 40
            add_manpower = 30
            swap_free_idea_group = yes
            swap_non_generic_missions = yes
            capital_scope = {
                build_to_forcelimit = {
                    infantry = 0.8
                    cavalry = 0.2
                }
            }
            FEZ = {
                every_owned_province = {
                    add_claim = ROOT
                }
            }
            SLE = {
                every_owned_province = {
                    add_claim = ROOT
                }
            }
            add_country_modifier = {
                name = armed_jihad_m
                duration = 7300
            }
		}
	}
}

country_event = {
    id = almoravids_events.4
    title = "almoravids_events.4.t"
    desc = "almoravids_events.4.d"
    picture = DIPLOMACY_eventPicture

    fire_only_once = yes
    is_triggered_only = yes

    option = {
        name = "almoravids_events.4.a"
        ai_chance = { factor = 100 }
        add_treasury = 200
        add_manpower = 10
        add_truce_with = FROM
        hidden_effect = {
            set_variable = { which = almoravid1_index value = 10 }
            while = {
                limit = {
                    check_variable = { which = almoravid1_index value = 1 }
                }
                subtract_variable = { which = almoravid1_index value = 1 }
                if = {
                    limit = { capital_scope = { controlled_by = ROOT } }
                    capital_scope = {
                        infantry = ROOT
                    }
                }
                else = {
                    random_owned_province = {
                        infantry = ROOT
                    }
                }
            }
        }
        add_country_modifier = {
            name = almoravid_assistance
            duration = -1
        }
    }
}

#MOR Intervention
country_event = {
    id = almoravids_events.6
    title = "almoravids_events.6.t"
    desc = "almoravids_events.6.d"
    picture = LAND_MILITARY_eventPicture

    fire_only_once = yes

    trigger = {
        has_global_flag = mor_leaves_andalus
        tag = MOR
        OR = {
            NOT = {
                any_subject_country = {
                    owns = 761
                }
            }
            any_neighbor_country = {
                is_free_or_tributary_trigger = yes
                religion_group = muslim
                capital_scope = {
                    OR = {
                        region = andalusia_region
                        region = alentejo_region
                        area = murcia_area
                    }
                }
            }
        }
    }

    mean_time_to_happen = {
        months = 12
    }

    immediate = {
        hidden_effect = {
            clr_global_flag = mor_leaves_andalus
            if = {
                limit = {
                    NOT = {
                        any_subject_country = {
                            owns = 761
                        }
                    }
                }
                761 = {
                    owner = {
                        save_event_target_as = mor_invasion_target
                    }
                }
            }
            else = {
                random_neighbor_country = {
                    limit = {
                        is_free_or_tributary_trigger = yes
                        religion_group = muslim
                        capital_scope = {
                            OR = {
                                region = andalusia_region
                                region = alentejo_region
                                area = murcia_area
                            }
                        }
                    }
                    save_event_target_as = mor_invasion_target
                }
            }
        }
    }

    option = {
        name = "almoravids_events.6.a"
        ai_chance = {
            factor = 90
            modifier = {
                factor = 0.5
                any_country = {
                    war_with = ROOT
                    war_score_against = {
                        who = ROOT
                        value = 25
                    }
                }
            }
            modifier = {
                factor = 0
                any_country = {
                    war_with = ROOT
                    war_score_against = {
                        who = ROOT
                        value = 50
                    }
                }
            }
            modifier = {
                factor = 0
                has_any_disaster = yes
            }
        }
        set_country_flag = almoravid_invasion_of_andalusia
        custom_tooltip = almoravids_events.tt.6
        event_target:mor_invasion_target = {
            ROOT = {
                declare_war_with_cb = {
                    casus_belli = cb_heretic
                    who = PREV
                }
            }
        }
        add_prestige = 10
        add_manpower = 10
        random_owned_province = {
            limit = {
                has_port = yes
                is_in_capital_area = yes
            }
            transport = ROOT
			transport = ROOT
			transport = ROOT
			transport = ROOT
			transport = ROOT
			transport = ROOT
			transport = ROOT
			transport = ROOT
			transport = ROOT
            transport = ROOT
        }
        hidden_effect = {
            event_target:mor_invasion_target = {
                country_event = {
                    id = almoravids_events.7
                }
            }
            every_country = {
                limit = {
                    capital_scope = {
                        OR = {
                            region = andalusia_region
                            region = alentejo_region
                            area = murcia_area
                        }
                    }
                    religion_group = muslim
                    NOT = { tag = ROOT }
                    NOT = { tag = event_target:mor_invasion_target }
                }
                country_event = {
                    id = almoravids_events.9
                }
            }
        }
    }
    option = {
        name = "almoravids_events.6.b"
        ai_chance = { factor = 10 }
        add_prestige = -10
    }
}

#Make MOR attack its neighbors
country_event = {
    id = almoravids_events.10
    title = "almoravids_events.10.t"
    desc = "almoravids_events.10.d"
    picture = SIEGE_eventPicture

    trigger = {
        normal_or_historical_nations = yes
        current_age = age_of_empire
        tag = MOR
        dynasty = "al-Murâbit"
        religion = sunni
        has_regency = no
        is_at_war = no
        any_neighbor_country = {
            religion_group = muslim
            NOT = { religion = sunni }
            capital_scope = {
                continent = africa
            }
            NOT = { is_subject_of = ROOT }
            NOT = { alliance_with = ROOT }
            NOT = { truce_with = ROOT }
            NOT = { guaranteed_by = ROOT }
        }
    }

    mean_time_to_happen = {
        months = 60
    }

    immediate = {
        hidden_effect = {
            random_neighbor_country = {
                limit = {
                    religion_group = muslim
                    NOT = { religion = sunni }
                    capital_scope = {
                        continent = africa
                    }
                    NOT = { is_subject_of = ROOT }
                    NOT = { alliance_with = ROOT }
                    NOT = { truce_with = ROOT }
                    NOT = { guaranteed_by = ROOT }
                }
                save_event_target_as = target_of_almoravids
            }
        }
    }

    option = {
        name = "almoravids_events.10.a"
        ai_chance = { factor = 75 }
        add_prestige = 10
        add_yearly_manpower = 1
        declare_war_with_cb = {
            casus_belli = cb_heretic
            who = event_target:target_of_almoravids
        }
    }
    option = {
        name = "almoravids_events.10.b"
        ai_chance = { factor = 25 }
        add_prestige = -5
    }
}
