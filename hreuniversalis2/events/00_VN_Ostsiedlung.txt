namespace = ostsiedlung

# culture convert province
province_event = {
	id = ostsiedlung.1
	title = ostsiedlung.1.t
	desc = ostsiedlung.1.d
	picture = COLONIZATION_eventPicture
	goto = root

	trigger = {
		normal_or_historical_nations = yes
		OR = {
			current_age = age_of_holy_wars
			current_age = age_of_crisis
		}
		OR = {
			culture = polabian
			culture = kashubian
			culture = pruthenian
			culture = sorbian
		}
		any_neighbor_province = {
			culture_group = germanic
		}
		owner = {
			primary_culture = polabian
			primary_culture = kashubian
			primary_culture = pruthenian
			primary_culture = sorbian
			culture_group = germanic
		}
	}

	mean_time_to_happen = {
		months = 300
		modifier = {
			factor = 0.5
			owner = {
				culture_group = germanic
			}
		}
		modifier = {
			factor = 1.5
			culture = schlesian
		}
	}

	option = {
		name = ostsiedlung.1.a
		ai_chance = {
			factor = 1
		}
		if = {
			limit = {
				culture = pruthenian
			}
			change_culture = prussian
		}
		else_if = {
			limit = {
				OR = {
					culture = sorbian
					culture = schlesian
				}
			}
			change_culture = prussian
		}
		else_if = {
			limit = {
				OR = {
					region = mecklenburg_region
					area = stolp_area
					area = kolberg_area
				}
			}
			change_culture = pommeranian
		}
		else_if = {
			limit = {
				OR = {
					region = brandenburg_region
					area = neumark_area
				}
			}
			change_culture = central_margravian
		}
		else_if = {
			limit = {
				OR = {
					region = holstein_region
				}
			}
			change_culture = low_elbish
		}
		else = {
			change_culture = prussian
		}
	}
	option = {
		name = ostsiedlung.1.b
		ai_chance = {
			factor = 0
		}
		add_province_modifier = {
			name = add_country_unrest_5_modifier
			duration = 3650
		}
	}
}
