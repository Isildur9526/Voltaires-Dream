		is_part_of_hre = yes
		any_known_country = {
			has_country_flag = poland_fragmentation_active
		}
		any_owned_province = {
			any_neighbor_province = {
				OR = {
					province_id = 50

				}

					OR = {
						primary_culture = polish
						primary_culture = mazovian
						primary_culture = schlesian
					}
				}
			}
		}
		NOT = {	owns = 50 }
		NOT = {	owns = 839 }
		NOT = {	owns = 242 }
		NOT = {	owns = 2197 }
		NOT = {	owns = 2650 }
		NOT = {	owns = 3205 }
		NOT = {	owns = 4917 }
		NOT = {	owns = 4805 }

			50 = { culture_group = germanic }
