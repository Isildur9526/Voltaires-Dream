estate_mamluks_hire_advisor = {
	can_select = {
		NOT = {
			employed_advisor = {
				category = ADM
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	task_requirements = {
		employed_advisor = {
			category = ADM
		}
		is_in_deficit = no
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_mamluks
			loyalty = 10
		}
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_mamluks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

		}

			set_country_flag = need_to_fire_adm_advisor

		clr_country_flag = need_to_fire_adm_advisor

			loyalty = 10
		}

		clr_country_flag = need_to_fire_adm_advisor
	}
	invalid_trigger = {
		NOT = {
			employed_advisor = {
				category = ADM
			}

	}

			duration = 7300
        }
		clr_country_flag = need_to_fire_adm_advisor
    }
}

#Increase autonomy in X
estate_mamluks_increase_autonomy_in_province_x = {
	can_select = {
		any_owned_province = {
			is_state_core = root
			NOT = { local_autonomy = 75 }
			NOT = { culture_group = root }
			is_capital = no
			NOT = { has_province_modifier = autonomy_increased }
			NOT = { has_province_modifier = autonomy_decreased }
		}
	}
	selection_weight = {
		factor = 1.5
	}
	pre_effect = {
		set_variable = {
			which = estate_mamluks_increase_autonomy_in_province_x_var
			value = 0
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { local_autonomy = 75 }
				NOT = { culture_group = root }
				is_capital = no
				NOT = { has_province_modifier = autonomy_increased }
				NOT = { has_province_modifier = autonomy_decreased }
			}
			save_event_target_as = agenda_province

			if = {
				limit = {
					local_autonomy = 60
				}
				root = {
					set_variable = {
						which = estate_mamluks_increase_autonomy_in_province_x_var
						value = 5
					}
				}
			}
			else_if = {
				limit = {
					local_autonomy = 45
				}
				root = {
					set_variable = {
						which = estate_mamluks_increase_autonomy_in_province_x_var
						value = 4
					}
				}
			}
			else_if = {
				limit = {
					local_autonomy = 30
				}
				root = {
					set_variable = {
						which = estate_mamluks_increase_autonomy_in_province_x_var
						value = 3
					}
				}
			}
			else_if = {
				limit = {
					local_autonomy = 15
				}
				root = {
					set_variable = {
						which = estate_mamluks_increase_autonomy_in_province_x_var
						value = 2
					}
				}
			}
			else = {
				root = {
					set_variable = {
						which = estate_mamluks_increase_autonomy_in_province_x_var
						value = 1
					}
				}
			}
			choose_random_development_bonus_effect = yes
		}
	}
	on_invalid = {
		event_target:agenda_province = {
			clear_random_development_bonus_flags_effect = yes
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			OR = {
				NOT = { owned_by = root }
				culture_group = root
				AND = {
					root = {
						check_variable = {
							which = estate_mamluks_increase_autonomy_in_province_x_var
							value = 5
						}
					}
					NOT = { local_autonomy = 50 }
				}
				AND = {
					root = {
						check_variable = {
							which = estate_mamluks_increase_autonomy_in_province_x_var
							value = 4
						}
					}
					NOT = { local_autonomy = 35 }
				}
				AND = {
					root = {
						check_variable = {
							which = estate_mamluks_increase_autonomy_in_province_x_var
							value = 3
						}
					}
					NOT = { local_autonomy = 20 }
				}
				AND = {
					root = {
						check_variable = {
							which = estate_mamluks_increase_autonomy_in_province_x_var
							value = 2
						}
					}
					NOT = { local_autonomy = 5 }
				}
			}
		}
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_mamluks_increase_autonomy_in_province_x_var
					value = 5
				}
			}
			event_target:agenda_province = {
				local_autonomy = 75
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = estate_mamluks_increase_autonomy_in_province_x_var
					value = 4
				}
			}
			event_target:agenda_province = {
				local_autonomy = 60
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = estate_mamluks_increase_autonomy_in_province_x_var
					value = 3
				}
			}
			event_target:agenda_province = {
				local_autonomy = 45
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = estate_mamluks_increase_autonomy_in_province_x_var
					value = 2
				}
			}
			event_target:agenda_province = {
				local_autonomy = 30
			}
		}
		else = {
			event_target:agenda_province = {
				local_autonomy = 15
			}
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_mamluks
			loyalty = 10
		}
		event_target:agenda_province = {
			apply_random_development_bonus_bonus_effect = yes
		}
    }
    failing_effect = {
		event_target:agenda_province = {
			clear_random_development_bonus_flags_effect = yes
		}
        add_estate_loyalty_modifier = {
            estate = estate_mamluks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Befriend X mamluks nation that is a neighbour
estate_mamluks_relations_with_X = {
	can_select = {
		any_owned_province = {
			is_state_core = root
			NOT = { culture_group = root }
			any_country = {
				culture_group = prev
				root = { knows_country = prev }
				is_neighbor_of = root
				NOT = { is_rival = root }
				ROOT = { NOT = { is_rival = prev } }
				OR = {
					is_free_or_tributary_trigger = yes
					is_subject_of = root
				}
				NOT = { war_with = root }
				NOT = { has_opinion = { who = root value = 50 } }
				has_opinion = { who = root value = -50 }
				NOT = {
					any_owned_province = {
						is_core = root
					}
				}
			}
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.75
			NOT = { num_of_free_diplomatic_relations = 1 }
		}
	}
	pre_effect = {
		set_variable = {
			which = estate_mamluks_relations_with_X
			value = 0
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { culture_group = root }
				any_country = {
					culture_group = prev
					root = { knows_country = prev }
					is_neighbor_of = root
					NOT = { is_rival = root }
					ROOT = { NOT = { is_rival = prev } }
					OR = {
						is_free_or_tributary_trigger = yes
						is_subject_of = root
					}
					NOT = { war_with = root }
					NOT = { has_opinion = { who = root value = 50 } }
					has_opinion = { who = root value = -50 }
					NOT = {
						any_owned_province = {
							is_core = root
						}
					}
				}
			}
			random_country = {
				limit = {
					culture_group = prev
					root = { knows_country = prev }
					is_neighbor_of = root
					NOT = { is_rival = root }
					ROOT = { NOT = { is_rival = prev } }
					OR = {
						is_free_or_tributary_trigger = yes
						is_subject_of = root
					}
					NOT = { war_with = root }
					NOT = { has_opinion = { who = root value = 50 } }
					has_opinion = { who = root value = -50 }
					NOT = {
						any_owned_province = {
							is_core = root
						}
					}
				}
				save_event_target_as = agenda_country
			}
		}
		if = {
			limit = {
				event_target:agenda_country = {
					NOT = { has_opinion = { who = root value = 0 } }
				}
			}
			set_variable = {
				which = estate_mamluks_relations_with_X
				value = 1
			}
		}
		else = {
			set_variable = {
				which = estate_mamluks_relations_with_X
				value = 2
			}
		}
	}
	fail_if = {
		event_target:agenda_country = {
			OR = {
				NOT = { num_of_cities = 1 }
				war_with = root
				NOT = { has_opinion = { who = root value = -100 } }
			}
		}
	}
	provinces_to_highlight = {
		is_capital_of = event_target:agenda_country
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_mamluks_relations_with_X
					value = 2
				}
			}
			event_target:agenda_country = {
				has_opinion = {
					who = root
					value = 100
				}
			}
		}
		else = {
			event_target:agenda_country = {
				has_opinion = {
					who = root
					value = 50
				}
			}
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_mamluks
			loyalty = 15
		}
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_mamluks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Take Slaves
estate_mamluks_take_slave_province = {
	can_select = {
		is_free_or_tributary_trigger = yes
		OR = {
			is_part_of_hre = no
			NOT = { hre_reform_passed = landfriede }
		}
		any_province = {
			trade_goods = slaves
			any_neighbor_province = { owned_by = ROOT }
			NOT = { country_or_non_sovereign_subject_holds = root }
			is_city = yes
			NOT = { owner = { ROOT = { truce_with = prev } } }
		}
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 0.5
			any_province = {
				trade_goods = slaves
				any_neighbor_province = { owned_by = ROOT }
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				owner = {
					NOT = { ROOT = { truce_with = prev } }
					alliance_with = root
				}
			}
		}
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_province = {
			limit = {
				trade_goods = slaves
				any_neighbor_province = { owned_by = ROOT }
				NOT = { country_or_non_sovereign_subject_holds = root }

				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
				NOT = { owner = { alliance_with = root } }
			}
			save_event_target_as = agenda_province
		}
		random_province = {
			limit = {
				trade_goods = slaves
				any_neighbor_province = { owned_by = ROOT }
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
				OR = {
					AND = {
						has_port = yes
						root = {
							any_owned_province = {
								has_port = yes
							}
						}
					}
					any_neighbor_province = {
						owned_by = root
					}
				}
			}
			save_event_target_as = agenda_province
		}
		random_province = {
			limit = {
				trade_goods = slaves
				any_neighbor_province = { owned_by = ROOT }
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }

					AND = {
						has_port = yes

							any_owned_province = {
								has_port = yes
							}
						}
					}
					any_neighbor_province = {
						owned_by = root
					}
				}
				NOT = { owner = { alliance_with = root } }
			}
			save_event_target_as = agenda_province

			is_subject = no
			NOT = { religion_group = ROOT }
			NOT = { truce_with = ROOT }
		}

		factor = 2
		modifier = {
			factor = 0.5
			any_neighbor_country = {
				is_subject = no
				alliance_with = ROOT
				NOT = { religion_group = ROOT }
				NOT = { truce_with = ROOT }
			}
		}
