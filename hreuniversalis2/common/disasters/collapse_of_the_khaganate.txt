collapse_of_the_khaganate = {
	potential = {
		OR = {
			tag = GLH
			tag = ILK
		}
		normal_or_historical_nations = yes
		is_free_or_tributary_trigger = yes
		NOT = { has_country_flag = had_collapse_of_the_khaganate }
		NOT = { current_age = age_of_reformation }
		NOT = { current_age = age_of_absolutism }
		NOT = { current_age = age_of_revolutions }
		NOT = { current_age = age_of_ideology }
	}


	can_start = {
		has_any_disaster = no
		num_of_cities = 100
		NOT = { legitimacy_equivalent = 50 }
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			legitimacy_equivalent = 50
			NOT = { num_of_cities = 100 }
		}
	}
		
	
	
	progress = {
		modifier = {
			factor = 1
			has_estate = estate_nomadic_tribes
			NOT = {
				estate_loyalty = {
					estate = estate_nomadic_tribes
					loyalty = 40
				}
			}
		}
		modifier = {
			factor = 1
			has_reform = steppe_horde
		}
		modifier = {
			factor = 1
			war_exhaustion = 5
		}
		modifier = {
			factor = 1
			war_exhaustion = 10
		}
		modifier = {
			factor = 0.5
			overextension_percentage = 0.33
		}		
		modifier = {
			factor = 1
			overextension_percentage = 0.66
		}
		modifier = {
			factor = 1
			overextension_percentage = 0.99
		}
		modifier = {
			factor = 0.5
			NOT = { legitimacy_equivalent = 50 }
		}
		modifier = {
			factor = 0.5
			NOT = { legitimacy_equivalent = 25 }
		}
		modifier = {
			factor = 1
			corruption = 5
		}
		modifier = {
			factor = 1
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 1
			average_unrest = 3
		}
		modifier = {
			factor = 1
			has_government_attribute = heir
			has_heir = no
		}
	}
	
	
	
	can_end = {
		NOT = { num_of_rebel_armies = 1 }
		NOT = { num_of_rebel_controlled_provinces = 1 }
		custom_trigger_tooltip = {
			tooltip = disaster_active_for_10_years_tooltip
			had_country_flag = {
				flag = had_collapse_of_the_khaganate
				days = 3650
			}
		}
	}
	

	modifier = {
		global_unrest = 10
		stability_cost_modifier = 0.5
		manpower_recovery_speed = -0.25
		reduced_liberty_desire = -25
	}	
	
	
	on_start = collapse_of_the_khaganate.1
	on_end = collapse_of_the_khaganate.100
	
	on_monthly = {
		events = {
		}
		random_events = { 
			1000 = 0
			100 = collapse_of_the_khaganate.2 #Trade Disputes
			100 = collapse_of_the_khaganate.3 #Decline of Warfare
			100 = collapse_of_the_khaganate.4 #Rising Corruption
			100 = collapse_of_the_khaganate.5 #Diplomatic Termination
			100 = collapse_of_the_khaganate.6 #Social Unrest
			# 7 triggered
		}
	}
}
