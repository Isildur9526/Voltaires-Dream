#Golden Bull of 1356
country_event = {
	id = vn_missions.203
	title = vn_missions.203.t
	desc = vn_missions.203.d
	picture = REFORM_eventPicture

	is_triggered_only = yes

    option = {
        name = "vn_missions.203.a"
		ai_chance = { factor = 90 }
		trigger = {
			is_elector = no
		}
        if = { 
            limit = {
				OR = {
					num_of_electors = 7
					num_of_electors = 8
				}
                any_elector = { 
					OR = {
						is_rival = ROOT 
						has_opinion = {
							who = ROOT
							value = 0
						}
					}
					is_subject = no
					was_player = no
				}
            }
            random_elector = {
                limit = { 
					OR = {
						is_rival = ROOT 
						has_opinion = {
							who = ROOT
							value = 0
						}
					}
					is_subject = no
					was_player = no
				}
				country_event = { id = vn_missions.204 }
				elector = no
            }
			custom_tooltip = vn_missions.203.a.tt
			set_country_flag = vn_missions_gained_electorate
			add_imperial_influence = 25
			elector = yes			
        }
		else_if = {
			limit = {
				OR = {
					num_of_electors = 7
					num_of_electors = 8
				}
				any_elector = { 
					is_subject = no
					was_player = no 				
                    NOT = {
						OR = {
							good_choice_for_electorship_trigger = yes
							has_opinion = {
								who = emperor
								value = 50
							}
                        }
                    }
                }
			}
            random_elector = {
                limit = { 
					is_subject = no
					was_player = no 				
                    NOT = {
						OR = {
							good_choice_for_electorship_trigger = yes
							has_opinion = {
								who = emperor
								value = 50
							}
                        }
                    }
				}
				country_event = { id = vn_missions.204 }
				elector = no
            }		
			custom_tooltip = vn_missions.203.a.tt
			set_country_flag = vn_missions_gained_electorate
			add_imperial_influence = 25
			elector = yes
		}
        else_if = {
			limit = {
				OR = {
					num_of_electors = 7
					num_of_electors = 8
				}
			}
			random_elector = { 
				country_event = { id = vn_missions.204 }
				elector = no
			}
			custom_tooltip = vn_missions.203.a.tt
			set_country_flag = vn_missions_gained_electorate
			add_imperial_influence = 25
			elector = yes
        }
		else = {
			custom_tooltip = vn_missions.203.ab.tt
			set_country_flag = vn_missions_gained_electorate
			add_imperial_influence = 25
			elector = yes			
		}
    }
	option = {
        name = "vn_missions.203.b"
		ai_chance = { factor = 10 }	
		trigger = {
			is_elector = no
		}
		add_prestige = -25
		add_legitimacy = -10
	}
	option = {
        name = "vn_missions.203.c"	
		trigger = {
			is_elector = yes
		}
		add_prestige = 25
		add_legitimacy = 10
		add_imperial_influence = 25
	}
}

#Excluded From the Electorate

	id = vn_missions.204
	title = vn_missions.204.t
	desc = vn_missions.204.d
	picture = REFORM_eventPicture

	is_triggered_only = yes

    option = {
        name = "vn_missions.204.a" #An Unfortunate Decision
		ai_chance = {
			factor = 25
		}
		random_elector = {
			limit = {
				has_country_flag = vn_missions_gained_electorate
			}
			add_prestige = -25
			reverse_add_opinion = {
				who = ROOT
				modifier = lux_golden_bull_opinion
			}
		}
    }
    option = {
        name = "vn_missions.204.b" #Fight for the Electorate
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0
				emperor = {
					army_strength = {
						who = ROOT
						value = 1
					}
				}
			}
		}
		random_elector = {
			limit = {
				has_country_flag = vn_missions_gained_electorate
			}
			reverse_add_opinion = {
				who = ROOT
				modifier = lux_golden_bull_opinion
			}
		}
		custom_tooltip = vn_missions.204.2.tt
		set_country_flag = vn_missions_fight_for_electorate
		add_casus_belli = {
			target = emperor
			type = cb_humiliate_rotw
			months = 24
		}
    }
}

#Defeat of the Emperor
country_event = {
	id = vn_missions.205
	title = vn_missions.205.t
	desc = vn_missions.205.d
	picture = REFORM_eventPicture

	fire_only_once = yes

	trigger = {
		has_country_flag = vn_missions_fight_for_electorate
		any_elector = {
			has_country_flag = vn_missions_gained_electorate
			NOT = {
				war_score_against = {
					who = ROOT
					value = -50
				}
			}
		}
	}
	
	mean_time_to_happen = {
 		days = 1
 	}	

    option = {
        name = "vn_missions.205.a" #Regain the Electorate 
		random_elector = {
			limit = {
				has_country_flag = vn_missions_gained_electorate
			}
			elector = no
		}		
		elector = yes
		add_prestige = 25
		add_country_modifier = {
			name = "lux_regained_electorate_mission"
			duration = -1
		}		
    }
}
