#The Church of the East
country_event = {
	id = vn_missions.10
	title = vn_missions.10.t
	desc = vn_missions.10.d
	picture = muslimgfx_WESTERNISATION_eventPicture

	is_triggered_only = yes

	option = {
		name = vn_missions.10.a
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				OR = {
					reverse_has_opinion = {
						who = PAP
						value = 50
					}
					NOT = {
						reverse_has_opinion = {
							who = PAP
							value = -50
						}
					}
				}
			}
		}
		trigger = { religion = nestorian }
		add_stability_or_adm_power = yes
		if = {
			limit = { has_dlc = "Mandate of Heaven" }
			if = {
				limit = { isolationism = 4 }
				reverse_add_opinion = {
					who = PAP
					modifier = opinion_displeased
				}
			}
			else_if = {
				limit = { isolationism = 3 }
				reverse_add_opinion = {
					who = PAP
					modifier = opinion_mildly_displeased
				}
			}
			else_if = {
				limit = { isolationism = 2 }
				reverse_add_opinion = {
					who = PAP
					modifier = opinion_amused
				}
			}
			else_if = {
				limit = { isolationism = 1 }
				reverse_add_opinion = {
					who = PAP
					modifier = opinion_pleased
				}
			}
			else = {
				reverse_add_opinion = {
					who = PAP
					modifier = opinion_good_relations
				}
			}
		}
		else = {
			add_prestige = 20
		}
	}
	option = {
		name = vn_missions.10.b
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = {
					reverse_has_opinion = {
						who = PAP
						value = 50
					}
				}
			}
		}
		trigger = { religion = nestorian }
		if = {
			limit = { has_dlc = "Mandate of Heaven" }
			if = {
				limit = {
					isolationism = 3
				}
				add_stability = -1
			}
			if = {
				limit = {
					isolationism = 1
				}
				add_isolationism = -1
			}
			else = {
				add_prestige = 25
			}
		}
		else = {
			add_legitimacy = 10
			add_prestige = 25
		}
		reverse_add_opinion = {
			who = PAP
			modifier = opinion_good_relations
		}
	}
	option = {
		name = vn_missions.10.c
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				reverse_has_opinion = {
					who = PAP
					value = -50
				}
			}
		}
		trigger = { religion = nestorian }
		if = {
			limit = { has_dlc = "Mandate of Heaven" }
			if = {
				limit = {
					NOT = { isolationism = 2 }
				}
				add_stability = -1
			}
			if = {
				limit = {
					isolationism = 4
				}
				add_isolationism = 1
			}
			else = {
				add_prestige = 25
			}
		}
		else = {
			add_legitimacy = 10
			add_prestige = 25
		}
		reverse_add_opinion = {
			who = PAP
			modifier = opinion_displeased
		}
	}
	option = {
		name = vn_missions.10.dd
		trigger = { NOT = { religion = nestorian } }
		add_prestige = 50
		add_stability_or_adm_power = yes
		hidden_effect = {
			6301 = {
				remove_province_triggered_modifier = theravada_kotte
			}
		}
		add_country_modifier = {
			name = religious_fanatics
			duration = 7300
		}
		every_province = {
			limit = { religion = nestorian }
			add_province_modifier = {
				name = add_unrest_5_modifier
				duration = 3650
			}
		}
	}
}
