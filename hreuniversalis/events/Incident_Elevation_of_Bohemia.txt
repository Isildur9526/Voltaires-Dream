country_event = {
	id = incident_elevation_of_bohemia.1
	title = "incident_elevation_of_bohemia.1.t"
	desc = "incident_elevation_of_bohemia.1.d"
	picture = GOOD_WITH_MONARCH_eventPicture

	fire_only_once = yes

		has_dlc = "Emperor"

		total_development = 300 #More than their 1054 level so they can't cheese it first month

		has_regency = no
		is_female = no
		NOT = { active_imperial_incident = any }
		is_elector = no
		emperor = {
			OR = {
				tag = BOH
				alliance_with = BOH
				has_opinion = {
					who = BOH
					value = 150
				}
			}
		}
		prestige = 0
		stability = 0
	}

	mean_time_to_happen = {
		months = 300
		modifier = {
			factor = 0.25
			NOT = { 
				OR = {
					num_of_electors = 7
					num_of_electors = 8
				}
			}
		}
	}
	
	option = {
		name = incident_elevation_of_bohemia.1.a
		if = {
			limit = {
				is_part_of_hre = yes
				is_emperor = no
			}
			set_imperial_incident = incident_elevation_of_bohemia
		}
		else_if = {
            limit = {
				OR = {
					num_of_electors = 7
					num_of_electors = 8
				}
                any_elector = { 
					OR = {
						is_rival = emperor 
						has_opinion = {
							who = emperor
							value = 0
						}
					}
					is_subject = no
					was_player = no
				}

                limit = { 
					OR = {
						is_rival = emperor 
						has_opinion = {
							who = emperor
							value = 0
						}
					}
					is_subject = no
					was_player = no
				}
				elector = no
            }
			elector = yes
			add_prestige = 20
			add_legitimacy = 20
			hidden_effect = {
				set_government_rank = 5
			}		
			custom_tooltip = boh_raised_to_kingdom_a_tt
        }
		else_if = {
			limit = {
				OR = {
					num_of_electors = 7
					num_of_electors = 8
				}
				any_elector = { 
					is_subject = no
					was_player = no 				
                    NOT = {

							good_choice_for_electorship_trigger = yes
							has_opinion = {

							}
                        }
                    }
                }
			}
            random_elector = {
                limit = {

                    NOT = {
						OR = {
							good_choice_for_electorship_trigger = yes

			}
		}

					who = emperor

			}

				set_country_flag = incident_elevation_of_bohemia_compromise
				country_event = { id = incident_elevation_of_bohemia.3 }
			}
		}
	}
	option = { #Reject A Change

		ai_chance = {

			modifier = {
