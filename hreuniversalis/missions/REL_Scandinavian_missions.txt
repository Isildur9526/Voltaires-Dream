VN_scandinavian_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		culture_group = scandinavian
		capital_scope = {
			superregion = scandinavia_superregion
		}
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = no
	
	vn_prove_religious_devotion = {
		icon = mission_religious
        required_missions = {} 
        position = 5
		trigger = {
			if = {
				limit = { religion = catholic }
				reverse_has_opinion = {
					who = PAP
					value = 150
				}
			}
			else_if = {
				limit = { religion = catholique }
				reverse_has_opinion = {
					who = AVI
					value = 150
				}
			}
			else_if = {
				limit = { religion = norse_pagan_reformed }
				if = {
					limit = { has_dlc = "Wealth of Nations" }
					OR = {
						has_personal_deity = odin
						has_personal_deity = freya
						has_personal_deity = tor
						has_personal_deity = tyr
						has_personal_deity = njord
						has_personal_deity = snotra
					}
				}
				any_neighbor_country = {
					NOT = { religion = ROOT }
					has_opinion = {
						who = ROOT
						value = 150
					}
				}
			}
			else = {
				num_of_allies = 5
			}
			religious_unity = 1
		}
		effect = {
			add_country_modifier = {
				name = "prove_religious_devotion_mission"
				duration = 9125
			}
		}
	}
	vn_enforce_feudalism = {
		icon = mission_noble_council
		required_missions = { vn_prove_religious_devotion } 
		trigger = {
			if = {
				limit = {
					has_estate = estate_nobles
				}
				has_estate_privilege = estate_nobles_land_rights
				num_of_estate_privileges = {
					estate = estate_nobles
					value = 3
				}
				estate_loyalty = {
					estate = estate_nobles
					loyalty = 60
				}
			}
			else = {
				stability = 3
			}
		}
		effect = {
			add_country_modifier = {
				name = "enforce_feudalism_mission"
				duration = 9125
			}
		}
	}
	vn_scandinavian_trade_fleet = {
		icon = mission_establish_high_seas_navy
		required_missions = { vn_enforce_feudalism }
		trigger = {
			navy_tradition = 35
			num_of_light_ship = 20

				value = 5

			}
		}
		effect = {
			add_country_modifier = {
				name = "scandinavian_trade_fleet_mission"
				duration = 9125
			}

				amount = 7

				5889 = { is_strongest_trade_power = ROOT }

				name = "north_sea_empire_mission"
